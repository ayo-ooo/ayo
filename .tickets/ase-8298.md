# Functional tests for agent creation and refinement

## Background

@ayo can dynamically create new agents when patterns emerge from user requests. It can also refine agents it created based on feedback and usage. These tests verify the agent lifecycle management works correctly.

## Why This Matters

Agent creation/refinement is a powerful feature that could go wrong:
- Creating duplicate agents
- Losing refinement history
- Incorrect capability inference after refinement
- Database inconsistency

func TestAgentCreation_Basic(t *testing.T) {
    manager, db := setupAgentManager(t)
    
    agent, err := manager.Create(CreateAgentRequest{
        Name:         "doc-writer",
        SystemPrompt: "You are a technical documentation writer.",
        Creator:      "@ayo",
        Reason:       "User frequently asked for documentation help",
    })
    
    assert.NoError(t, err)
    assert.Equal(t, "doc-writer", agent.Name)
    assert.Equal(t, "You are a technical documentation writer.", agent.SystemPrompt)
    
    // Verify in database
    var count int
    db.QueryRow("SELECT COUNT(*) FROM ayo_created_agents WHERE agent_id = ?", agent.ID).Scan(&count)
    assert.Equal(t, 1, count)
}

func TestAgentCreation_InvalidName(t *testing.T) {
    manager, _ := setupAgentManager(t)
    
    testCases := []struct{
        name string
        err  string
    }{
        {"", "name required"},
        {"has.dot", "invalid character"},
        {"has space", "invalid character"},
        {"UPPERCASE", "lowercase only"},
        {strings.Repeat("a", 100), "too long"},
    }
    
    for _, tc := range testCases {
        _, err := manager.Create(CreateAgentRequest{
            Name:         tc.name,
            SystemPrompt: "Test",
            Creator:      "@ayo",
        })
        
        assert.Error(t, err, "Expected error for name: %s", tc.name)
        assert.Contains(t, err.Error(), tc.err)
    }
}

func TestAgentRefinement_UpdatePrompt(t *testing.T) {
    manager, db := setupAgentManager(t)
    
    // Create agent
    agent, _ := manager.Create(CreateAgentRequest{
        Name:         "helper",
        SystemPrompt: "Original prompt",
        Creator:      "@ayo",
    })
    
    // Refine it
    refined, err := manager.Refine(agent.ID, RefineRequest{
        NewSystemPrompt: "Improved prompt with better instructions",
        Reason:          "User feedback indicated original was unclear",
    })
    
    assert.NoError(t, err)
    assert.Equal(t, "Improved prompt with better instructions", refined.SystemPrompt)
    
    // Verify refinement recorded
    var count int
    db.QueryRow("SELECT refinement_count FROM ayo_created_agents WHERE agent_id = ?", agent.ID).Scan(&count)
    assert.Equal(t, 1, count)
}

func TestAgentRefinement_History(t *testing.T) {
    manager, _ := setupAgentManager(t)
    
    agent, _ := manager.Create(CreateAgentRequest{
        Name:         "evolving",
        SystemPrompt: "Version 1",
        Creator:      "@ayo",
    })
    
    // Multiple refinements
    manager.Refine(agent.ID, RefineRequest{NewSystemPrompt: "Version 2", Reason: "First improvement"})
    manager.Refine(agent.ID, RefineRequest{NewSystemPrompt: "Version 3", Reason: "Second improvement"})
    
    // Get history
    history, err := manager.GetRefinementHistory(agent.ID)
    
    assert.NoError(t, err)
    assert.Len(t, history, 2)
    assert.Equal(t, "First improvement", history[0].Reason)
    assert.Equal(t, "Second improvement", history[1].Reason)
}

func TestAgentRefinement_CannotRefineBuiltin(t *testing.T) {
    manager, _ := setupAgentManager(t)
    
    // Try to refine @ayo (builtin)
    _, err := manager.Refine("builtin-ayo", RefineRequest{
        NewSystemPrompt: "Hacked prompt",
        Reason:          "Attack",
    })
    
    assert.Error(t, err)
    assert.Contains(t, err.Error(), "cannot refine builtin")
}

func TestAgentRefinement_CapabilitiesRecomputed(t *testing.T) {
    manager, db := setupAgentManager(t)
    
    // Create general agent
    agent, _ := manager.Create(CreateAgentRequest{
        Name:         "general",
        SystemPrompt: "General purpose assistant",
        Creator:      "@ayo",
    })
    
    // Get original capability count
    var origCount int
    db.QueryRow("SELECT COUNT(*) FROM agent_capabilities WHERE agent_id = ?", agent.ID).Scan(&origCount)
    
    // Refine to be specialized
    manager.Refine(agent.ID, RefineRequest{
        NewSystemPrompt: "Expert security auditor specializing in code review",
        Reason:          "User needed security expertise",
    })
    
    // Capabilities should be different
    var newCaps []string
    rows, _ := db.Query("SELECT name FROM agent_capabilities WHERE agent_id = ?", agent.ID)
    for rows.Next() {
        var name string
        rows.Scan(&name)
        newCaps = append(newCaps, name)
    }
    
    assert.Contains(t, newCaps, "security-analysis")
}

func TestAgentPromotion(t *testing.T) {
    manager, db := setupAgentManager(t)
    
    agent, _ := manager.Create(CreateAgentRequest{
        Name:         "promoted",
        SystemPrompt: "Test",
        TrustLevel:   "sandboxed",
        Creator:      "@ayo",
    })
    
    err := manager.Promote(agent.ID, "privileged")
    
    assert.NoError(t, err)
    
    // Verify updated
    updated, _ := manager.Get(agent.ID)
    assert.Equal(t, "privileged", updated.TrustLevel)
}

func TestAgentArchival(t *testing.T) {
    manager, db := setupAgentManager(t)
    
    agent, _ := manager.Create(CreateAgentRequest{
        Name:         "archived",
        SystemPrompt: "Test",
        Creator:      "@ayo",
    })
    
    err := manager.Archive(agent.ID, "No longer needed")
    
    assert.NoError(t, err)
    
    // Should not appear in normal list
    agents := manager.List()
    for _, a := range agents {
        assert.NotEqual(t, agent.ID, a.ID)
    }
    
    // But should exist with archived flag
    var archived bool
    db.QueryRow("SELECT is_archived FROM ayo_created_agents WHERE agent_id = ?", agent.ID).Scan(&archived)
    assert.True(t, archived)
}
