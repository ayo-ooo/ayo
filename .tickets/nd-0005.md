func selectSandboxProvider() providers.SandboxProvider {
    switch runtime.GOOS {
    case "darwin":
        if p := NewAppleProvider(); p.IsAvailable() {
            return p
        }
    case "linux":
        if p := NewLinuxProvider(); p.IsAvailable() {
            return p
        }
    }
    return NewNoneProvider()
}
