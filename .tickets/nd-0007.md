## File to Modify

### install.sh

Update the Apple Container setup:

```bash
# Detect if Apple Container is available
check_apple_container() {
    # Must be macOS
    if [ "$(uname -s)" != "Darwin" ]; then
        return 1
    fi
    
    # Must be Apple Silicon
    if [ "$(uname -m)" != "arm64" ]; then
        echo "Note: Apple Container requires Apple Silicon (arm64)"
        return 1
    fi
    
    # Check macOS version (26+)
    local version
    version=$(sw_vers -productVersion | cut -d. -f1)
    if [ "$version" -lt 26 ]; then
        echo "Note: Apple Container requires macOS 26 or later (you have $version)"
        return 1
    fi
    
    # Check if container command exists
    if ! command -v container &> /dev/null; then
        return 1
    fi
    
    return 0
}

setup_apple_container() {
    echo "Setting up Apple Container..."
    
    # Check if already installed
    if check_apple_container; then
        echo "Apple Container is available"
        
        # Start the container service if not running
        if ! container system status &>/dev/null; then
            echo "Starting Apple Container service..."
            container system start || {
                echo "Warning: Could not start Apple Container service"
                echo "You may need to run: container system start"
            }
        fi
        
        # Pull base image if needed
        if ! container image list 2>/dev/null | grep -q "busybox"; then
            echo "Pulling busybox base image..."
            container image pull busybox:stable || {
                echo "Warning: Could not pull busybox image"
            }
        fi
        
        return 0
    fi
    
    # Not available - provide installation guidance
    echo ""
    echo "Apple Container is not available on this system."
    echo ""
    echo "Requirements:"
    echo "  - macOS 26 or later"
    echo "  - Apple Silicon (M1/M2/M3)"
    echo "  - Apple Container tool installed"
    echo ""
    echo "To install Apple Container:"
    echo "  1. Download from: https://github.com/apple/container/releases"
    echo "  2. Run the installer package"
    echo "  3. Start the service: container system start"
    echo ""
    echo "Continuing without container support (using host execution)..."
    echo ""
    
    return 1
}

# Main setup logic
setup_sandbox() {
    case "$(uname -s)" in
        Darwin)
            setup_apple_container
            ;;
        Linux)
            echo "Linux container support coming soon"
            echo "Using host execution for now"
            ;;
        *)
            echo "Unsupported platform: $(uname -s)"
            echo "Using host execution"
            ;;
    esac
}
```

Remove:
- `setup_docker()` function entirely
- Any calls to `setup_docker`
- Docker-related checks
