## Objective

## Context

After the Docker removal and Apple Container updates, we need robust tests that:
1. Test provider interface compliance
2. Test mock provider for CI environments
3. Test provider selection logic
4. Test graceful fallback behavior

### internal/integration/provider_integration_test.go (NEW FILE)

Integration tests for real providers:

```go
package integration

import (
    "context"
    "runtime"
    "testing"
    "time"

    "github.com/alexcabrera/ayo/internal/providers"
    "github.com/alexcabrera/ayo/internal/sandbox"
)

func TestAppleProvider_Integration(t *testing.T) {
    if runtime.GOOS != "darwin" {
        t.Skip("Apple Container requires Darwin")
    }
    
    p := sandbox.NewAppleProvider()
    if !p.IsAvailable() {
        t.Skip("Apple Container not available")
    }
    
    ctx, cancel := context.WithTimeout(context.Background(), 60*time.Second)
    defer cancel()
    
    // Create sandbox
    sb, err := p.Create(ctx, providers.SandboxCreateOptions{
        Name:  "test-" + t.Name(),
        Image: "busybox:stable",
    })
    if err != nil {
        t.Fatal(err)
    }
    defer p.Delete(ctx, sb.ID, true)
    
    // Exec command
    result, err := p.Exec(ctx, sb.ID, providers.ExecOptions{
        Command: "echo hello",
    })
    if err != nil {
        t.Fatal(err)
    }
    
    if result.ExitCode != 0 {
        t.Errorf("exit code = %d, want 0", result.ExitCode)
    }
    if !strings.Contains(result.Stdout, "hello") {
        t.Errorf("stdout = %q, want to contain 'hello'", result.Stdout)
    }
}
```
